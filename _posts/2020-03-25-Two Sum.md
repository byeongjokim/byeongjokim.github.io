```python
def twoSum(self, nums: List[int], target: int) -> List[int]:
    new_nums = [(nums[i], i) for i in range(len(nums))]
    new_nums.sort(key=lambda x:x[0])
    
    i = 0
    j = len(nums)-1
    
    while(True):
        now_sum = new_nums[i][0] + new_nums[j][0]
        if  now_sum < target:
            i = i + 1
        if now_sum > target:
            j = j - 1
        if now_sum == target:
            return [new_nums[i][1], new_nums[j][1]]
```
